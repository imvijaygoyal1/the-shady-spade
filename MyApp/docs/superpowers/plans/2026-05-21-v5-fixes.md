### CRIT-03 — joinSession TOCTOU — wrap in Firestore transaction

- [ ] **Step 3: Rewrite joinSession slot-claim as a transaction**

Replace the current `getDocument` + `updateData` pattern in `joinSession` (lines ~232–290) with a Firestore transaction. The transaction atomically reads the session, validates slot availability, and writes the claim:

```swift
func joinSession(code: String, uid: String, name: String, avatar: String = "") async throws {
    let db = Firestore.firestore()
    let ref = db.collection("sessions").document(code)

    var resolvedJoinIndex: Int = -1

    // Atomically claim a slot — prevents two simultaneous joins from both
    // reading the same empty slot and overwriting each other.
    try await db.runTransaction { transaction, errorPointer in
        let snapshot: DocumentSnapshot
        do {
            snapshot = try transaction.getDocument(ref)
        } catch let error as NSError {
            errorPointer?.pointee = error
            return nil
        }

        guard snapshot.exists,
              let data = snapshot.data(),
              let slotsData = data["playerSlots"] as? [[String: Any]] else {
            errorPointer?.pointee = NSError(
                domain: "JoinSession", code: -1,
                userInfo: [NSLocalizedDescriptionKey: "Session not found"])
            return nil
        }

        let sessionStatus = data["status"] as? String ?? "waiting"
        guard sessionStatus == "waiting" else {
            errorPointer?.pointee = NSError(
                domain: "JoinSession", code: -2,
                userInfo: [NSLocalizedDescriptionKey: "Session already started"])
            return nil
        }

        let rawAISeats = (data["aiSeats"] as? [Any] ?? [])
            .compactMap { ($0 as? Int) ?? ($0 as? Int64).map(Int.init) }
            .sorted()

        let joinIndex: Int
        if let firstAI = rawAISeats.first, firstAI >= 0 && firstAI < slotsData.count {
            joinIndex = firstAI
        } else {
            guard let idx = slotsData.firstIndex(where: { !($0["joined"] as? Bool ?? false) }) else {
                errorPointer?.pointee = NSError(
                    domain: "JoinSession", code: -3,
                    userInfo: [NSLocalizedDescriptionKey: "No slots available"])
                return nil
            }
            joinIndex = idx
        }

        var updated = slotsData
        updated[joinIndex] = ["uid": uid, "name": name, "avatar": avatar, "joined": true]
        let newAISeats = rawAISeats.filter { $0 != joinIndex }
        transaction.updateData(["playerSlots": updated, "aiSeats": newAISeats], forDocument: ref)
        resolvedJoinIndex = joinIndex
        return nil
    }

    guard resolvedJoinIndex >= 0 else {
        throw URLError(.resourceUnavailable)
    }

    sessionCode = code
    isHost = false
    myJoinedSlotIndex = resolvedJoinIndex
    attachListener(code: code)
}
```

Note: `db.runTransaction` returns `Any?` — ignore the return value (it's `nil` by convention). The `errorPointer` mechanism propagates NSErrors as Swift throws.

### MED-05 — Firestore Security Rules

In Firebase Console → Firestore → Rules, add/update the rules to scope `pendingAction` writes to registered session players:

```javascript
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    match /sessions/{sessionId} {
      // Anyone authenticated can read a session (needed for lobby display)
      allow read: if request.auth != null;

      // Only the host (slot 0) can create a session
      allow create: if request.auth != null
                    && request.resource.data.hostUid == request.auth.uid;

      // Allow writes to pendingAction only from registered session players
      allow update: if request.auth != null
                    && isRegisteredPlayer(request.auth.uid, resource.data);

      function isRegisteredPlayer(uid, data) {
        return data.playerSlots != null
               && data.playerSlots.hasAny([{'uid': uid, 'joined': true}]);
      }
    }

    // All other collections — adjust as needed
    match /{document=**} {
      allow read, write: if request.auth != null;
    }
  }
}
```

Note: Test these rules in the Firebase Console Rules Playground before deploying. Ensure the host (who writes `gameState`) also passes `isRegisteredPlayer`.

### LOW-07 — UserDefaults migration layer

- [ ] **Step 1: Add a migration check in MyAppApp.swift**

Add a one-time migration check at app launch. This handles the case where `PendingGameRecord` schema changes in the future:

```swift
// In MyAppApp.swift, in the App init or onAppear of the root view:
func migrateUserDefaultsIfNeeded() {
    let currentSchemaVersion = 1  // increment when PendingGameRecord changes
    let storedVersion = UserDefaults.standard.integer(forKey: "leaderboard_schema_version")
    guard storedVersion < currentSchemaVersion else { return }

    // Migration: if the stored records can't be decoded with the current schema,
    // clear them rather than crash. Records are offline-queued — losing them is
    // preferable to a crash on launch.
    if let data = UserDefaults.standard.data(forKey: "leaderboard_pending_records_v1") {
        if (try? JSONDecoder().decode([PendingGameRecord].self, from: data)) == nil {
            UserDefaults.standard.removeObject(forKey: "leaderboard_pending_records_v1")
        }
    }

    UserDefaults.standard.set(currentSchemaVersion, forKey: "leaderboard_schema_version")
}
```

Call `migrateUserDefaultsIfNeeded()` early in app startup (before `LeaderboardService.shared` is first accessed).
